// 升序
function sort_arr(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
      }
    }
  }
  return arr
}
// 降序

function sort_arr_opt(arr) {
  let tag = true
  for (let i = 0; i < arr.length - 1; i++) {
    if (!tag) break
    tag = false
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
        tag = true
      }
    }
  }
  return arr
}

// 排序算法2 快速排序 最优nlogn 最差n^2
function quick_sort(arr) {
  if (arr.length <= 1) return arr
  let left = [],
    right = []
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] < arr[0]) {
      left.push(arr[i])
    } else {
      right.push(arr[i])
    }
  }
  return quick_sort(left).concat(arr[0], quick_sort(right))
}
